def required_chol(age, probability, intercept, coef_age, coef_chol):
    log_odds = np.log(probability / (1 - probability))
    chol = (log_odds - intercept - coef_age * age) / coef_chol
    return chol

age = 55
chol_50_per = required_chol(age, 0.5, intercept, coef_age, coef_chol)
chol_90_per = required_chol(age, 0.9, intercept, coef_age, coef_chol)

2. Linear Discriminant Analysis

Fit an LDA model using only age and chol (cholesterol) as predictors.

For a 55 year old, how high would their cholesterol need to be for the doctors to predict heart disease is present?

367.87 is the chol level needed for the doctor to predict .50 chance of heart disease and 1587.72 is the chol level needed for the doctor to predict .90 chance of heart disease

4. Comparing Decision Boundaries

Make a scatterplot of age and chol, coloring the points by their true disease outcome. Add a line to the plot representing the linear separator (aka decision boundary) for each of the three models above.

import pandas as pd
import numpy as np
from plotnine import ggplot, aes, geom_point, geom_line, theme_minimal, labs

# Generate a range of age values for decision boundaries
age_range = np.linspace(X['age'].min(), X['age'].max(), 100)

# Calculate cholesterol range for decision boundaries
chol_range_lr = -(intercept + coef_age * age_range) / coef_chol  # Logistic Regression
chol_range_lda = -(lda_intercept + lda_coef_age * age_range) / lda_coef_chol  # LDA
chol_range_svc = -(svc_intercept + svc_coef_age * age_range) / svc_coef_chol  # SVC

# Create a DataFrame for decision boundaries
boundary_data = pd.DataFrame({
    'age': age_range,
    'chol_lr': chol_range_lr,
    'chol_lda': chol_range_lda,
    'chol_svc': chol_range_svc
})

# Convert boundary data to long format for plotnine
boundary_l = pd.melt(boundary_data, id_vars='age',
                        var_name='model', value_name='chol')

# Combine predictors and target into a single DataFrame
plot_data = X.copy()
plot_data['diagnosis'] = y

# Create the plot
plot = (
    ggplot(plot_data, aes(x='age', y='chol', color='factor(diagnosis)'))
    + geom_point(alpha=0.7)
    + geom_line(boundary_l, aes(x='age', y='chol', color='model'), size=1)
    + theme_minimal()
    + labs(
        title='Comparison of Decision Boundaries',
        x='Age',
        y='Cholesterol (mg/dL)',
        color='Legend\n(0=No Disease, 1=Disease)'
    )
)
