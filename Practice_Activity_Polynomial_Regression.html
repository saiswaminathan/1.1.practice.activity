Let’s use bill_length_mm to predict bill_depth_mm. Prepare your data and fit the following models on the entire dataset:

  • Simple linear regression (e.g. straight-line) model
  • Quadratic (degree 2 polynomial) model
  • Cubic (degree 3 polynomial) model
  • Degree 10 polynomial model

Make predictions for each model and plot your fitted models on the scatterplot.

import numpy as np
import pandas as pd
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import PolynomialFeatures
from sklearn.metrics import mean_squared_error
X = penguins[['bill_length_mm']].copy()
y = penguins['bill_depth_mm'].values
X_1 = X.copy()
X_2 = X.copy()
X_2['x_sq'] = X_2['bill_length_mm'] ** 2
X_3 = X.copy()
X_3['x_sq'] = X_3['bill_length_mm'] ** 2
X_3['x_cube'] = X_3['bill_length_mm'] ** 3
X_10 = X.copy()
X_10['x_10'] = X_10['bill_length_mm'] ** 10
models = {
    'Degree 1': LinearRegression().fit(X_1, y),
    'Degree 2': LinearRegression().fit(X_2, y),
    'Degree 3': LinearRegression().fit(X_3, y),
    'Degree 10': LinearRegression().fit(X_10, y)
}
x_range = np.linspace(X['bill_length_mm'].min(), X['bill_length_mm'].max(), 500)
results = pd.DataFrame({'bill_length_mm': x_range})

for degree in models.keys():
    if degree == 'Degree 1':
        results[f'{degree}_Prediction'] = models[degree].predict(x_range.reshape(-1, 1))
    elif degree == 'Degree 2':
        x_range_poly = pd.DataFrame({'bill_length_mm': x_range})
        x_range_poly['x_sq'] = x_range_poly['bill_length_mm'] ** 2
        results[f'{degree}_Prediction'] = models[degree].predict(x_range_poly)
    elif degree == 'Degree 3':
        x_range_poly = pd.DataFrame({'bill_length_mm': x_range})
        x_range_poly['x_sq'] = x_range_poly['bill_length_mm'] ** 2
        x_range_poly['x_cube'] = x_range_poly['bill_length_mm'] ** 3
        results[f'{degree}_Prediction'] = models[degree].predict(x_range_poly)
    elif degree == 'Degree 10':
        x_range_poly = pd.DataFrame({'bill_length_mm': x_range})
        x_range_poly['x_10'] = x_range_poly['bill_length_mm'] ** 10
        results[f'{degree}_Prediction'] = models[degree].predict(x_range_poly)

print("\nSample Predictions:")
print(results.head())

from sklearn.metrics import mean_squared_error
print("\nModel Performance Metrics (Mean Squared Error):")
for degree, model in models.items():
    if degree == 'Degree 1':
        mse = mean_squared_error(y, model.predict(X_1))
    elif degree == 'Degree 2':
        mse = mean_squared_error(y, model.predict(X_2))
    elif degree == 'Degree 3':
        mse = mean_squared_error(y, model.predict(X_3))
    elif degree == 'Degree 10':
        mse = mean_squared_error(y, model.predict(X_10))
    print(f"{degree}: MSE = {mse:.3f}")

import matplotlib.pyplot as plt
X = penguins[['bill_length_mm']].values
y = penguins['bill_depth_mm'].values

# Scatterplot of the original data
plt.figure(figsize=(10, 6))
plt.scatter(X, y, color='blue', alpha=0.6, edgecolors='k', label="Original Data")

# Fit and plot models of different polynomial degrees
degrees = [1, 2, 3, 10]
colors = ['red', 'green', 'orange', 'purple']

for i, degree in enumerate(degrees):
    # Polynomial feature transformation
    poly = PolynomialFeatures(degree=degree)
    X_poly = poly.fit_transform(X)
    
    # Fit linear regression model
    model = LinearRegression()
    model.fit(X_poly, y)
    
    # Make predictions
    y_pred = model.predict(X_poly)
    
    # Plot the fitted model
    plt.plot(X, y_pred, color=colors[i], label=f"Degree {degree} (MSE: {mean_squared_error(y, y_pred):.2f})")

# Add plot details
plt.xlabel("Bill Length (mm)")
plt.ylabel("Bill Depth (mm)")
plt.title("Polynomial Regression Models")
plt.legend()
plt.show()
